impl<T> LinkClone for T

    T: 'static + Linkable+ LinkAppend + Clone,

    fn clone_box(&self) -> Box<dyn Linkable> {
        Box::new(self.clone())
    }

    fn clone(&self) -> Box<dyn Linkable> {
        self.clone_box()
    }
